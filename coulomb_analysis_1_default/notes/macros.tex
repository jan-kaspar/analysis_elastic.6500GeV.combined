\def\parseGet#1_#2;{%
	\def\parseCurr{#1}%
	\def\parseBuf{#2}%
}

\def\parseLoop{%
	\expandafter\parseGet\parseBuf;%
	\ifx\parseOutput\empty
		\edef\parseOutput{\parseCurr}%
	\else
		\edef\parseOutput{\parseOutput\_\parseCurr}%
	\fi
	\ifx\parseBuf\empty
	\else
		\parseLoop
	\fi
}

\def\parse#1{%
	\def\parseBuf{#1_}%
	\def\parseOutput{}%
	\parseLoop
	\parseOutput
}

\gdef\plot#1{%
	\leavevmode
	\pdfstartlink user {%
	    /Subtype /Link
	    /A << 
	        /Type /Action 
	        /S /URI 
	        /URI (\baseDir/plots/#1) 
	    >>}%
	\cBlue [\parse{#1}]\cBlack
	\pdfendlink
}
